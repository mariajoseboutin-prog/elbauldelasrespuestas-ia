from flask import Flask, request, jsonify
import openai
import os

app = Flask(__name__)

# Usa la clave API desde las variables de entorno (Render)
openai.api_key = os.getenv("OPENAI_API_KEY")

@app.route('/')
def inicio():
    return "âœ… API con IA funcionando en Flask"

@app.route('/api/chat', methods=['POST'])
def chat():
    data = request.get_json()
    prompt = data.get('prompt', '')

    if not prompt:
        return jsonify({'error': 'Falta el prompt'}), 400

    try:
        response = openai.Completion.create(
            engine="text-davinci-003",  # Puedes cambiar al modelo que uses
            prompt=prompt,
            max_tokens=150,
            temperature=0.7
        )
        answer = response.choices[0].text.strip()
        return jsonify({'respuesta': answer})
    except Exception as e:
        return jsonify({'error': str(e)}), 500

# Render necesita que usemos su puerto
if __name__ == '__main__':
    port = int(os.environ.get("PORT", 5000))
    app.run(host="0.0.0.0", port=port)
